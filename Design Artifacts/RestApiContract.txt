//Create a user account
POST
/public/User
Payload - 
{
	"name":"profileName",
	"userName":"userName",
	"pwd":"pasWord"
}

Response
{
	"id":"userID"
}

//GET a user account details for display
GET
/public/User/{userID}
Payload - 
{
	"name":"profileName",
	"userName":"userName",
}

//Edit a user account
PUT
/public/User/{userID}
Payload - 
{
	"name":"profileName",
	"userName":"userName",
	"pwd":"pasWord"
}

Response
{
	"id":"userID"
}

//Validate a user for LogIn
POST
/public/User/{userName}
//Sending password in header will be considered while implementation.
Payload - 
{
	"pwd":"pasWord"
}

Response
{
	"id":"userID",
	"jwtToken":"someToken"
}

//Create a Blog
POST
/public/User/{userName}/Blog
Payload - 
{
	"title":"blogTitle",
	"content":"blogContent"
}

Edit an existing Blog
PUT
/public/User/{userName}/Blog/{blogID}
Payload - 
{
	"title":"blogTitle",
	"content":"blogContent"
}

//Get latest Blogs for Home Page. This API supports search also using query parameters.
GET
/public/Blog?title=someTitle&content=someContent
Response - 
{
	"contents": [
		{
			"id":"blogID1",
			"title":"blogTitle1",
			"content":"blogContent1",
			"publishTime": "publishTime1"
		},
		{
			"id":"blogID2",
			"title":"blogTitle2",
			"content":"blogContent2",
			"publishTime": "publishTime2"
		}
	],
	"next": "/public/Blog?Locator=11"
}

		}
